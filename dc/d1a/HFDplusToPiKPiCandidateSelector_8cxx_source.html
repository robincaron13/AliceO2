65  template <typename T>
66  int getpTBin(T candpT)
67  {
68  double pTBins[npTBins + 1] = {1., 2., 3., 4., 5., 6., 7., 8., 10., 12., 16., 24., 36.};
69  if (candpT < pTBins[0] || candpT >= pTBins[npTBins]) {
70  return -1;
71  }
72  for (int i = 0; i < npTBins; i++) {
73  if (candpT < pTBins[i + 1]) {
74  return i;
75  }
76  }
77  return -1;
78  }

99  template <typename T1, typename T2>
100  bool selection(const T1& hfCandProng3, const T2& trackPion1, const T2& trackKaon, const T2& trackPion2)
101  {
102  auto candpT = hfCandProng3.pt();
103  int pTBin = getpTBin(candpT);
104  if (pTBin == -1) {
105  return false;
106  }
107  if (candpT < d_pTCandMin || candpT > d_pTCandMax) {
108  return false; //check that the candidate pT is within the analysis range
109  }
110  if (trackPion1.pt() < cuts[pTBin][1] || trackKaon.pt() < cuts[pTBin][2] || trackPion2.pt() < cuts[pTBin][1]) {
111  return false; // cut on daughter pT
112  }
113  if (TMath::Abs(InvMassDPlus(hfCandProng3) - RecoDecay::getMassPDG(411)) > cuts[pTBin][0]) {
114  return false; // invariant mass cut
115  }
116  if (hfCandProng3.decayLength() < cuts[pTBin][3]) {
117  return false;
118  }
119  if (hfCandProng3.decayLengthXYNormalised() < cuts[pTBin][4]) {
120  return false;
121  }
122  if (hfCandProng3.cpa() < cuts[pTBin][5]) {
123  return false;
124  }
125  if (hfCandProng3.cpaXY() < cuts[pTBin][6]) {
126  return false;
127  }
128  if (hfCandProng3.maxNormalisedDeltaIP() > cuts[pTBin][7]) {
129  return false;
130  }
131  return true;
132  }

169  template <typename T>
170  bool selectionPIDTPC(const T& track, int nPDG, int nSigmaCut)
171  {
172  double nSigma = 100.0; //arbitarily large value
173  nPDG = TMath::Abs(nPDG);
174  if (nPDG == 211) {
175  nSigma = track.tpcNSigmaPi();
176  } else if (nPDG == 321) {
177  nSigma = track.tpcNSigmaKa();
178  } else {
179  return false;
180  }
181  return nSigma < nSigmaCut;
182  }

210  template <typename T>
211  int selectionPID(const T& track, int nPDG)
212  {
213  int statusTPC = -1; //no PID info
214  int statusTOF = -1; //no PID info
215 
216  if (validTPCPID(track)) {
217  if (!selectionPIDTPC(track, nPDG, d_nSigmaTPC)) {
218  statusTPC = 0; //rejected by PID
219  } else {
220  statusTPC = 1; //positive PID
221  }
222  }
223 
224  if (validTOFPID(track)) {
225  if (!selectionPIDTOF(track, nPDG, d_nSigmaTOF)) {
226  statusTOF = 0; //rejected by PID
227  } else {
228  statusTOF = 1; //positive PID
229  }
230  }
231 
232  //conservative PID strategy
233  if (statusTPC == 1 && statusTOF != 0) {
234  return 1;
235  } else if (statusTPC == 0 || statusTOF == 0) {
236  return 0;
237  } else {
238  return -1;
239  }
240  }

242  void process(aod::HfCandProng3 const& hfCandProng3s, aod::BigTracksPID const& tracks)
243  {
244  int statusDplusToPiKPi; // final selection flag : 0-rejected 1-accepted
245  int pionPlus1, kaonMinus, pionPlus2;
246 
247  for (auto& hfCandProng3 : hfCandProng3s) { //looping over 3 prong candidates
248 
249  statusDplusToPiKPi = 0;
250 
251  if (!(hfCandProng3.hfflag() & 1 << DPlusToPiKPi)) {
252  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
253  continue;
254  }
255 
256  auto trackPos1 = hfCandProng3.index0_as<aod::BigTracksPID>(); //positive daughter (negative for the antiparticles)
257  auto trackNeg1 = hfCandProng3.index1_as<aod::BigTracksPID>(); //negative daughter (positive for the antiparticles)
258  auto trackPos2 = hfCandProng3.index2_as<aod::BigTracksPID>(); //positive daughter (negative for the antiparticles)
259 
260  // daughter track validity selection
261  if (!daughterSelection(trackPos1) || !daughterSelection(trackNeg1) || !daughterSelection(trackPos2)) {
262  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
263  continue;
264  }
265 
266  // topological selection
267  if (!selection(hfCandProng3, trackPos1, trackNeg1, trackPos2)) {
268  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
269  continue;
270  }
271 
272  // pid selection
273  pionPlus1 = selectionPID(trackPos1, 211);
274  kaonMinus = selectionPID(trackNeg1, 321);
275  pionPlus2 = selectionPID(trackPos2, 211);
276 
277  if (pionPlus1 == 0 || kaonMinus == 0 || pionPlus2 == 0) { //exclude Dplus for PID
278  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
279  continue;
280  }
281 
282  statusDplusToPiKPi = 1;
283  hfSelDplusToPiKPiCandidate(statusDplusToPiKPi);
284  }
285  }

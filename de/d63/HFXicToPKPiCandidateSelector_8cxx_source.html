101  if (pTBin == -1) {
102  return false;

104

106  return false; //check that the candidate pT is within the analysis range

129  template <typename T1, typename T2>
130  bool selectionTopolConjugate(const T1& hfCandProng3, const T2& trackProton, const T2& trackKaon, const T2& trackPion)
131  {
132 
133  auto candpT = hfCandProng3.pt();
134  int pTBin = getpTBin(candpT);
135  if (pTBin == -1) {
136  return false;
137  }
138 
139  if (trackProton.pt() < cuts[pTBin][1] || trackKaon.pt() < cuts[pTBin][2] || trackPion.pt() < cuts[pTBin][3]) {
140  return false; //cut on daughter pT
141  }
142 
143  if (trackProton.globalIndex() == hfCandProng3.index0Id()) {
144  if (TMath::Abs(InvMassXicToPKPi(hfCandProng3) - RecoDecay::getMassPDG(4232)) > cuts[pTBin][0]) {
145  return false;
146  }
147  } else {
148  if (TMath::Abs(InvMassXicToPiKP(hfCandProng3) - RecoDecay::getMassPDG(4232)) > cuts[pTBin][0]) {
149  return false;
150  }
151  }
152 
153  return true;
154  }
